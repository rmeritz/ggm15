def post(self, request, *args, **kwargs):
    form_class = self.get_form_class()
    form = self.get_form(form_class)
    if form.is_valid():
        return self.form_valid(form)
    else:
        return self.form_invalid(form)

jobs/forms.py 

                              
from datetime import timedelta
from django import forms
from .models import Job


class JobCreateForm(forms.ModelForm):
    class Meta:
        model = Job
        fields = ["name", "description", "start_date"]

    def clean_name(self):
        name = self.cleaned_data.get('name')
        if self._is_obscene(name):
            raise forms.ValidationError(
                'You may not choose an obscene name')
        return name

    def save(self, *args, **kwargs):
        kwargs.update({'commit': False})
        job = super(JobCreateForm, self).save(*args, **kwargs)
        job.end_date = job.start_date + timedelta(days=30)
        job.save
        return job

lib/validators/obscene_validator.rb 

                              
class ObsceneValidator < ActiveModel::Validator
  def validate(record)
    if record.name.obscene?
      record.errors[:name] << 'No dirty names!'
    end
  end
end

Reflections

  • Django allow different validations for different user

Reflections

  • Django's Managers allow the models to have a cleaner API
  • Rails' schema is centralized
  • Django's migrations are an imperfect solution to migration conflicts
  • Both muddle the difference btw DB and non-DB level validations
